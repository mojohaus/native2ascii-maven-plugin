17  package org.codehaus.mojo.native2ascii.mojo;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.util.Iterator;
22  
23  import org.apache.maven.plugin.MojoExecutionException;
24  import org.apache.maven.plugins.annotations.LifecyclePhase;
25  import org.apache.maven.plugins.annotations.Mojo;
26  import org.apache.maven.plugins.annotations.Parameter;
27  import org.apache.maven.shared.utils.StringUtils;
28  import org.codehaus.mojo.native2ascii.Native2Ascii;
29  
30  /**
31   * Converts files with characters in any supported character encoding to one with ASCII and/or
32   * Unicode escapes.
33   */
34  @Mojo(name = "resources", defaultPhase = LifecyclePhase.PROCESS_RESOURCES)
35  public class Native2AsciiMojo extends AbstractNative2AsciiMojo {
36  
37    /**
38     * Target directory.
39     */
40    @Parameter(defaultValue = "${project.build.outputDirectory}")
41    public File targetDir;
42  
43    /**
44     * Source directory.
45     */
46    @Parameter(defaultValue = "src/main/native2ascii")
47    public File srcDir;
48  
49  
50    @Override
51    protected File getSourceDirectory() {
52      return srcDir;
53    }
54  
55  
56    /**
57     * @return the target directory
58     */
59    protected File getTargetDirectory() {
60      return targetDir;
61    }
62  
63  
64    @Override
65    public void executeTransformation(final Iterator<File> files) throws MojoExecutionException {
66      if (!getTargetDirectory().exists()) {
67        getTargetDirectory().mkdirs();
68      }
69  
70      while (files.hasNext()) {
71        final File file = files.next();
72        getLog().info("Processing " + file.getAbsolutePath());
73        try {
74          new Native2Ascii(getLog()).nativeToAscii(file, getTargetFile(file), encoding);
75        } catch (final IOException e) {
76          throw new MojoExecutionException("Unable to convert " + file.getAbsolutePath(), e);
77        }
78      }
79    }
80  
81  
82    private File getTargetFile(final File sourceFile) throws IOException {
83      return new File(StringUtils.replace(sourceFile.getCanonicalPath(), getSourceDirectory().getCanonicalPath(),
84          getTargetDirectory().getCanonicalPath()));
85    }
86  }
